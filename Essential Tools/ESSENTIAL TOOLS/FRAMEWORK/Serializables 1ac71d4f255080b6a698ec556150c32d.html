Info:

Serializable Action paramless an class to permit serialize UnityActions into saving and load methods.

View:

How To Use: 

	  #region Variables

		public UnityAction unityAction;

		public SerializableAction serializableAction;
		
		#endregion
		
		#region UnityMethods
		
		public void ExampleTest()
		{
				unityAction += ExampleTest;
	
				serializableAction = new SerializableAction(this, "ExampleTest");
    }
		
		#endregion

Conversion: 

	// FROM
	serializableAction = SerializableAction.ConvertFromUnityAction(unityAction, this);
	 
	// TO
	unityAction = SerializableAction.ConvertToUnityAction(serializableAction);

ðŸ”¸SerializableScriptableObject
ðŸ”¸SerializableScriptableObject

Info:

Serializable Scriptable Object an class to permit serialize Object as Scriptable Object. (interface only, require coding)

How To Use: 

[CreateAssetMenu(fileName = "MyData", menuName = "ScriptableObjects/MyData")]
public class MyData : SerializableScriptableObject<MyData>
{
    public int myInt;
    public string myString;
    public Vector3 myVector;

    public override void OnDeserialization(object sender)
    {
        base.OnDeserialization(sender);
        Debug.Log("MyData Deserialized");
    }
}

public class MyDataSaver : MonoBehaviour
{
    public MyData dataToSave;
    public string fileName = "myData.json";

    public void SaveData()
    {
        if (dataToSave != null)
        {
            dataToSave.Serialize(fileName, dataToSave);
            Debug.Log("Data saved to " + fileName);
        }
    }
}

public class MyDataLoader : MonoBehaviour
{
    public MyData dataToLoad;
    public string fileName = "myData.json";

    public void LoadData()
    {
        if (dataToLoad != null && File.Exists(fileName))
        {
            dataToLoad.Deserialize(fileName);
            Debug.Log("Data loaded from " + fileName);
        }
    }
}
