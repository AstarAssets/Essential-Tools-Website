ðŸ”¸How to Custom Configurate Editor

Info:

Each component of the Framework Core features a dedicated custom editor. To extend its functionality for your specific needs, follow the guidelines below

View:

Functions:

  • override SetName()
  • override SetIcon()
  • override DrawValues()
  • override DrawIcons()

Examples: 

using AstarLibrary.Core.Framework;
using AstarLibrary.AstarEditor.Customs;

public class Example : Element // Here you can use any part like Cog, Manager, Module etc..
{
    public bool iconButton; // Some parts will show instant only by having fields or properties here.
}


[CustomEditor(typeof(Example), true)]
public class ExampleCustomEditor : ElementCustomEditor
{
    #region Variables
    protected SerializedProperty _toggleMyIcon; // Pointer for a boolean in Example Class

    public Texture2D _iconInfo;

    public string TOOLTIP_INFO = "my Tooltip";
    #endregion

    #region Public Methods
    public override void OnEnable()
    {
        base.OnEnable();

        //Load your textures (icons) properties here.
        _iconInfo = AssetDatabase.LoadAssetAtPath<Texture2D>(AssetUtility.HandleFindAsset("Default"));
    }

    public override void Setup()
    {
        base.Setup();

        // Your Code here | Recommend initalize Textures, Properties etc.. here.
    }
    public override void SetName()
    {
        _elementName = "Quick Actions"; // _[name of part]Name for change the title of QuickActions
    }
    public override void SetIcon()
    {
        _elementIcon = "Essential-Tools_Icon_Unique_Chrono"; // _[name of part]Icon for change the icon of Part, string is case sensitive
    }
    public override void DrawIcons()
    {
        base.DrawIcons();

        // If you need more Icons put then like this to stay in the pattern, or chose clear and put on your way
        _toggleMyIcon.boolValue = GUILayout.Toggle(_toggleMyIcon.boolValue, new GUIContent(_iconInfo, TOOLTIP_INFO), "Button", GUILayout.Width(20), GUILayout.Height(20));
    }
    public override void DrawBody()
    {
        base.DrawBody();

        //Your body custom here.
    }
    #endregion
}
